create table art( #艺术作品表
  	id int auto_increment primary key comment '主键 自增',
  	user_id int not null comment '用户id 不为空',
  	name varchar(200) not null comment '作品名称 不为空',
  	state int not null comment '状态 不为空 0 待审核 1 审核通过 2 审核失败',
  	kind varchar(100) comment '艺术品分类 可为空',
  	reason varchar(500) comment '审核失败原因 可为空',
    photo varchar(500) not null comment '作品主图 不为空',
  	photo0 varchar(500) not null comment '作者与作品合影 不为空',
  	photo1 varchar(500) comment '作品其他方位图片1 可为空',
  	photo2 varchar(500) comment '作品其他方位图片2 可为空',
  	photo3 varchar(500) comment '作品其他方位图片3 可为空',
  	photo4 varchar(500) comment '作品局部特征图片 可为空',
  	gongyi varchar(100) not null comment '主要工艺 不为空',
  	type varchar(100) not null comment '造型种类 不为空',
  	length float not null comment '长 不为空',
  	width float not null comment '宽 不为空',
  	height float not null comment '高 不为空',
  	build_date varchar(8) not null comment '创作时间 不为空',
  	title varchar(100) not null comment '题款 不为空',
  	introduction varchar(500) comment '作品介绍 可为空',
  	location_bit char(10) not null comment '位置位图 10位 不为空 每一位1和0表示 默认0000000000',
    create_date varchar(8) not null comment '创建日期 不为空',
    create_time varchar(6) not null comment '创建时间 不为空',
    create_ip varchar(100) not null comment '创建IP 不为空'
)ENGINE=InnoDB DEFAULT CHARSET=utf8;
